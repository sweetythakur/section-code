<script>
      $(document).ready(function(){

        setTimeout(function(){
      
      
      var trigger_tab_data = localStorage.getItem('popup');
      
      
      console.log(trigger_tab_data);
      
          if( trigger_tab_data != 'show_popup'){
      
      $('.popup-overlay').fadeIn();
      
      $('.popup').fadeIn();
        
        localStorage.setItem('popup', 'show_popup' );
        
      }
      
    }, 3000);
        
      
      });
    </script>
